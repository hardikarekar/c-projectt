## Selection statements
# if	
	Syntax:-
	if ( expression )  
	{
		statement1;
		...  
	}
	else  // optional
	{
		statement2;
		...
	}

	if (Exp1) 
	{
		// stbe if Exp1 is true
	}
	else if(Exp2) 
	{
		// stbe if Exp1 is false and Exp2 is true
	}
	else if (Exp3) 
	{
		// stbe if Exp1,Exp2 is false 
			and Exp3 is true
	}
	.
	.
	else 
	{
		// stbe if all exp are false
	}
